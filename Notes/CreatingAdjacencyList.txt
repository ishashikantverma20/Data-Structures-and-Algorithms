
n = 5
edges = [[0, 1], [0, 2], [0, 3], [1, 4]]

 0 -> [1,2,3]
 1 -> [0,4]
 2 -> [0]
 3 -> [0]
 4 -> [1]


Map<Integer, List<Integer>> graph = new HashMap<>();

for (int i = 0; i < n; i++) {
    graph.put(i, new ArrayList<>());
}

for (int[] edge : edges) {
    graph.get(edge[0]).add(edge[1]);
    graph.get(edge[1]).add(edge[0]); // for undirected graph both directions cover 0 -> 1 and 1 -> 0
}
